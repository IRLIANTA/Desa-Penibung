Basic Usage

// Metode 1: Menggunakan shorthand methods
Toast.success('Data berhasil disimpan!');
Toast.warning('Periksa kembali data Anda.');
Toast.danger('Terjadi kesalahan sistem!');
Toast.info('Update tersedia untuk aplikasi.');

// Metode 2: Menggunakan method utama
Toast.show('Pesan custom', 'success', 5000);

// Parameters:
// message: string - Pesan yang akan ditampilkan
// type: 'success'|'warning'|'danger'|'info' - Tipe toast
// duration: number - Durasi dalam milisekon (0 = manual close)

HTML Integration

<!-- 1. Tambahkan container toast -->
<div id="toast-container" class="fixed top-4 right-4 z-50 space-y-2 max-w-sm w-full"></div>

<!-- 2. Panggil dari button atau event -->
<button onclick="Toast.success('Berhasil!')">Success</button>
<button onclick="Toast.warning('Peringatan!')">Warning</button>

<!-- 3. Gunakan dalam form -->
<form onsubmit="handleSubmit(event)">
  <!-- form fields -->
</form>

<script>
function handleSubmit(e) {
  e.preventDefault();
  // Process form...
  Toast.success('Form berhasil dikirim!');
}
</script>
